# 데이터 모델 — 전체 엔티티 정의

---

## 1. User (유저)

```typescript
interface User {
  id: string
  email: string
  nickname: string          // 최대 20자
  profileImage?: string
  bio?: string              // 최대 100자
  role: 'user' | 'admin' | 'editor'

  // 소셜 연동
  googleId?: string
  kakaoId?: string

  // 통계 (캐시)
  followersCount: number
  followingCount: number
  gamesCount: number        // 컬렉션 게임 수
  playsCount: number        // 플레이 기록 수

  // 설정
  notificationSettings: NotificationSettings
  privacySettings: PrivacySettings

  createdAt: Date
  updatedAt: Date
}
```

## 2. Game (보드게임)

```typescript
interface Game {
  id: string
  bggId?: string            // BoardGameGeek ID (연동용)

  // 기본 정보
  titleKo: string           // 한글 이름
  titleEn: string           // 영문 이름
  titleAlias?: string[]     // 별칭 (검색용)

  // 설명
  description?: string
  shortDescription?: string // 한 줄 설명

  // 게임 스펙
  minPlayers: number
  maxPlayers: number
  minPlayTime?: number      // 분
  maxPlayTime?: number
  minAge?: number
  complexity?: number       // 1~5 복잡도

  // 분류
  genres: string[]          // 전략, 파티, 가족, ...
  mechanics: string[]       // 덱 빌딩, 경매, 협력, ...
  themes: string[]          // 판타지, SF, 역사, ...

  // 발매 정보
  designer?: string[]
  artist?: string[]
  publisher?: string[]
  releaseYear?: number

  // 미디어
  thumbnail: string         // 메인 커버 이미지
  images?: string[]         // 추가 이미지

  // 집계 (캐시)
  avgRating?: number
  ratingCount: number
  playsCount: number        // 커뮤니티 전체 플레이 수
  collectorsCount: number   // 보유자 수

  // 가격/구매
  msrp?: number             // 권장 소비자가
  purchaseLinks?: PurchaseLink[]

  // 상태
  status: 'active' | 'discontinued' | 'upcoming'
  isFeatured: boolean

  createdAt: Date
  updatedAt: Date
}

interface PurchaseLink {
  store: string             // 판매처명
  url: string
  price: number
  inStock: boolean
  updatedAt: Date
}
```

## 3. Collection (컬렉션)

type CollectionStatus =

## 4. PlayLog (플레이 기록)

```typescript
interface PlayLog {
  id: string
  userId: string
  gameId: string
  game: Game

  // 플레이 정보
  playedAt: Date
  durationMinutes?: number  // 실제 플레이 시간

  // 참가자
  playerCount: number
  players: PlayLogPlayer[]  // 함께한 사람

  // 장소
  location?: string
  localId?: string          // 로컬 장소 연동

  // 결과
  isWin?: boolean           // 협동/솔로 게임의 경우 승리 여부
  winner?: string           // 승자 닉네임
  scores?: PlayScore[]      // 점수 기록

  // 평가
  rating?: number           // 1~5
  memo?: string

  // 미디어
  images?: string[]

  // 공개 설정
  isPublic: boolean

  createdAt: Date
  updatedAt: Date
}

interface PlayLogPlayer {
  name: string
  userId?: string           // 앱 유저면 연동
  score?: number
  isWinner: boolean
}

interface PlayScore {
  playerName: string
  score: number
}
```

## 8. Local (로컬 장소)

```typescript
type LocalType = 'cafe' | 'store' | 'meetup_space'

interface Local {
  id: string
  registeredBy: string      // userId

  name: string
  type: LocalType
  description?: string

  // 위치
  address: string
  lat: number
  lng: number
  region: string            // 시/도
  district: string          // 구/군

  // 정보
  phone?: string
  website?: string
  sns?: string

  // 운영시간
  hours?: BusinessHours[]

  // 요금
  entryFee?: number
  hourlyFee?: number

  // 게임 보유 목록
  games?: Game[]

  // 미디어
  thumbnail?: string
  images?: string[]

  // 집계
  avgRating?: number
  reviewCount: number

  status: 'active' | 'closed' | 'pending'

  createdAt: Date
  updatedAt: Date
}

interface BusinessHours {
  day: 0 | 1 | 2 | 3 | 4 | 5 | 6  // 0=일요일
  openTime: string    // "10:00"
  closeTime: string   // "22:00"
  isClosed: boolean
}
```

type UsedType = 'sell' | 'buy' | 'trade' | 'share'

}

interface Meetup {

  // 일정/장소

  maxParticipants: number
  currentParticipants: number
  participants: MeetupParticipant[]

  fee?: number
  level: MeetupLevel

  status: 'recruiting' | 'full' | 'closed' | 'completed'

  updatedAt: Date
}

interface MeetupParticipant {
  userId: string
  user: User
  status: 'pending' | 'approved' | 'rejected'

}

## 13. Follow / Notification

```typescript
interface Follow {
  id: string
  followerId: string        // 팔로우 하는 사람
  followingId: string       // 팔로우 받는 사람
  createdAt: Date
}

type NotificationType =
  | 'price_drop'            // 가격 하락
  | 'used_game_listed'      // 원하는 게임 중고 등록
  | 'post_comment'          // 내 글 댓글
  | 'post_like'             // 내 글 좋아요
  | 'new_follower'          // 새 팔로워
  | 'meetup_approved'       // 모임 신청 수락
  | 'meetup_reminder'       // 모임 D-1
  | 'system'                // 시스템

interface Notification {
  id: string
  userId: string
  type: NotificationType
  title: string
  body: string
  link?: string
  isRead: boolean
  createdAt: Date
}
```
