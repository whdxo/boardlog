# DB 스키마 설계 — Supabase (PostgreSQL)

> 기존 `03_데이터_모델.md`의 TypeScript 인터페이스를 PostgreSQL 테이블로 변환.
> Supabase Auth의 `auth.users`를 활용하고, `public.profiles`로 확장한다.

---

## 전체 ERD 요약

```
auth.users (Supabase 관리)
    │
    └── profiles (1:1)
          │
          ├── collections (1:N) ──── games (N:1)
          ├── ratings (1:N) ──────── games (N:1)
          ├── play_logs (1:N) ────── games (N:1)
          │     └── play_log_players (1:N)
          ├── selections (1:N)
          │     └── selection_games (N:M) ── games
          ├── posts (1:N)
          │     └── comments (1:N, self-ref)
          ├── local_places (1:N)
          │     └── local_reviews (1:N)
          ├── used_posts (1:N)
          ├── follows (N:M, self-ref)
          └── notifications (1:N)
```

## 1. profiles (유저 프로필)

Supabase Auth의 `auth.users`를 기본 인증으로 사용하고, 추가 정보는 `profiles`에 저장.

```sql
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  email TEXT NOT NULL,
  nickname TEXT NOT NULL UNIQUE,
  profile_image TEXT,
  bio TEXT,
  role TEXT NOT NULL DEFAULT 'user' CHECK (role IN ('user', 'admin', 'editor')),

  -- 소셜 연동 (Supabase Auth가 관리하지만 조회용 캐시)
  google_id TEXT,
  kakao_id TEXT,

  -- 통계 캐시 (트리거로 갱신)
  followers_count INTEGER NOT NULL DEFAULT 0,
  following_count INTEGER NOT NULL DEFAULT 0,
  games_count INTEGER NOT NULL DEFAULT 0,
  plays_count INTEGER NOT NULL DEFAULT 0,

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- 닉네임 검색용 인덱스
CREATE INDEX idx_profiles_nickname ON profiles (nickname);

-- 새 유저 가입 시 자동으로 profiles 행 생성 (트리거)
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO profiles (id, email, nickname)
  VALUES (
    NEW.id,
    NEW.email,
    COALESCE(NEW.raw_user_meta_data->>'nickname', 'user_' || LEFT(NEW.id::text, 8))
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();
```

## 2. games (보드게임)

```sql
CREATE TABLE games (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  -- 기본 정보
  title_ko TEXT NOT NULL,
  title_en TEXT,
  title_alias TEXT[],                -- 별칭 (검색용)
  description TEXT,
  short_description TEXT,

  -- 스펙
  min_players INTEGER NOT NULL DEFAULT 1,
  max_players INTEGER NOT NULL DEFAULT 4,
  min_play_time INTEGER,             -- 분
  max_play_time INTEGER,
  min_age INTEGER,
  complexity NUMERIC(2,1),           -- 1.0 ~ 5.0

  -- 분류
  genres TEXT[] NOT NULL DEFAULT '{}',
  mechanics TEXT[] NOT NULL DEFAULT '{}',
  themes TEXT[] NOT NULL DEFAULT '{}',

  -- 발매 정보
  designers TEXT[],
  artists TEXT[],
  publishers TEXT[],
  release_year INTEGER,

  -- 미디어
  thumbnail TEXT,
  images TEXT[],

  -- 집계 캐시
  avg_rating NUMERIC(3,1) DEFAULT 0,
  rating_count INTEGER NOT NULL DEFAULT 0,
  plays_count INTEGER NOT NULL DEFAULT 0,
  collectors_count INTEGER NOT NULL DEFAULT 0,

  -- 가격/구매
  msrp INTEGER,                      -- 권장소비자가 (원)

  -- 외부 연동
  bgg_id TEXT,

  -- 상태
  status TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'discontinued', 'upcoming')),
  is_featured BOOLEAN NOT NULL DEFAULT false,

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- 검색용 인덱스
CREATE INDEX idx_games_title_ko ON games USING gin (to_tsvector('simple', title_ko));
CREATE INDEX idx_games_genres ON games USING gin (genres);
CREATE INDEX idx_games_status ON games (status);
CREATE INDEX idx_games_avg_rating ON games (avg_rating DESC);
CREATE INDEX idx_games_release_year ON games (release_year DESC);
```

## 4. collections (컬렉션)

```sql
CREATE TABLE collections (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  game_id UUID NOT NULL REFERENCES games(id) ON DELETE CASCADE,

  status TEXT NOT NULL CHECK (status IN (
    'owned', 'fan', 'played', 'wishlist', 'preorder', 'funded', 'selling', 'sold'
  )),

  note TEXT,
  purchased_at DATE,
  purchase_price INTEGER,

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),

  -- 한 유저가 같은 게임에 같은 상태를 중복 등록 불가
  UNIQUE (user_id, game_id)
);

CREATE INDEX idx_collections_user ON collections (user_id);
CREATE INDEX idx_collections_user_status ON collections (user_id, status);
CREATE INDEX idx_collections_game ON collections (game_id);
```

```sql
  short_review TEXT,

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),

  UNIQUE (user_id, game_id)
);

CREATE INDEX idx_ratings_game ON ratings (game_id);
CREATE INDEX idx_ratings_user ON ratings (user_id);
```

## 10. local_places (로컬 장소)

```sql
CREATE TABLE local_places (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  registered_by UUID NOT NULL REFERENCES profiles(id),

  name TEXT NOT NULL,
  type TEXT NOT NULL CHECK (type IN ('cafe', 'store', 'meetup_space')),
  description TEXT,

  -- 위치
  address TEXT NOT NULL,
  lat DOUBLE PRECISION NOT NULL,
  lng DOUBLE PRECISION NOT NULL,
  region TEXT,                       -- 시/도
  district TEXT,                     -- 구/군

  -- 연락처
  phone TEXT,
  website TEXT,
  sns TEXT,

  -- 요금
  entry_fee INTEGER,
  hourly_fee INTEGER,

  -- 미디어
  thumbnail TEXT,
  images TEXT[],

  -- 집계 캐시
  avg_rating NUMERIC(2,1) DEFAULT 0,
  review_count INTEGER NOT NULL DEFAULT 0,

  status TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'closed', 'pending')),

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- 운영시간
CREATE TABLE business_hours (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  place_id UUID NOT NULL REFERENCES local_places(id) ON DELETE CASCADE,

  day_of_week SMALLINT NOT NULL CHECK (day_of_week BETWEEN 0 AND 6), -- 0=일
  open_time TIME,
  close_time TIME,
  is_closed BOOLEAN NOT NULL DEFAULT false
);

-- 장소 리뷰
CREATE TABLE local_reviews (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  place_id UUID NOT NULL REFERENCES local_places(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,

  score NUMERIC(2,1) NOT NULL CHECK (score >= 1 AND score <= 5),
  content TEXT NOT NULL,
  images TEXT[],

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),

  UNIQUE (place_id, user_id)
);

CREATE INDEX idx_local_places_type ON local_places (type);
CREATE INDEX idx_local_places_region ON local_places (region, district);
CREATE INDEX idx_local_reviews_place ON local_reviews (place_id);
```

## 11. shop_products (쇼핑 상품)

```sql
CREATE TABLE shop_products (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  game_id UUID REFERENCES games(id),

  name TEXT NOT NULL,
  description TEXT,
  thumbnail TEXT NOT NULL,

  original_price INTEGER NOT NULL,
  sale_price INTEGER NOT NULL,
  discount_rate NUMERIC(4,1) GENERATED ALWAYS AS (
    CASE WHEN original_price > 0
      THEN ROUND((1 - sale_price::NUMERIC / original_price) * 100, 1)
      ELSE 0
    END
  ) STORED,

  store_name TEXT NOT NULL,
  store_url TEXT NOT NULL,

  in_stock BOOLEAN NOT NULL DEFAULT true,
  stock_count INTEGER,

  category TEXT NOT NULL DEFAULT 'game' CHECK (category IN ('game', 'accessory')),

  view_count INTEGER NOT NULL DEFAULT 0,
  wish_count INTEGER NOT NULL DEFAULT 0,

  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_shop_products_game ON shop_products (game_id);
CREATE INDEX idx_shop_products_category ON shop_products (category);
CREATE INDEX idx_shop_products_sale_price ON shop_products (sale_price);
```

## 15. RLS (Row Level Security) 정책

모든 테이블에 RLS를 활성화하고, 기본 정책을 설정한다.

```sql
-- 모든 테이블에 RLS 활성화
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE collections ENABLE ROW LEVEL SECURITY;
ALTER TABLE ratings ENABLE ROW LEVEL SECURITY;
ALTER TABLE play_logs ENABLE ROW LEVEL SECURITY;
ALTER TABLE posts ENABLE ROW LEVEL SECURITY;
ALTER TABLE comments ENABLE ROW LEVEL SECURITY;
ALTER TABLE follows ENABLE ROW LEVEL SECURITY;
ALTER TABLE notifications ENABLE ROW LEVEL SECURITY;
ALTER TABLE used_posts ENABLE ROW LEVEL SECURITY;
ALTER TABLE selections ENABLE ROW LEVEL SECURITY;
ALTER TABLE local_reviews ENABLE ROW LEVEL SECURITY;
ALTER TABLE post_likes ENABLE ROW LEVEL SECURITY;

-- games, shop_products, local_places는 공개 데이터 → 읽기만 허용
ALTER TABLE games ENABLE ROW LEVEL SECURITY;
ALTER TABLE shop_products ENABLE ROW LEVEL SECURITY;
ALTER TABLE local_places ENABLE ROW LEVEL SECURITY;

-- ────────────────────────────────────────────
-- 공개 데이터: 누구나 읽기 가능
-- ────────────────────────────────────────────
CREATE POLICY "games: 누구나 읽기" ON games FOR SELECT USING (true);
CREATE POLICY "shop_products: 누구나 읽기" ON shop_products FOR SELECT USING (true);
CREATE POLICY "local_places: 누구나 읽기" ON local_places FOR SELECT USING (true);

-- ────────────────────────────────────────────
-- profiles
-- ────────────────────────────────────────────
CREATE POLICY "profiles: 누구나 읽기" ON profiles FOR SELECT USING (true);
CREATE POLICY "profiles: 본인만 수정" ON profiles FOR UPDATE USING (auth.uid() = id);

-- ────────────────────────────────────────────
-- collections
-- ────────────────────────────────────────────
CREATE POLICY "collections: 누구나 읽기" ON collections FOR SELECT USING (true);
CREATE POLICY "collections: 본인만 생성" ON collections FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "collections: 본인만 수정" ON collections FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "collections: 본인만 삭제" ON collections FOR DELETE USING (auth.uid() = user_id);

-- ────────────────────────────────────────────
-- posts
-- ────────────────────────────────────────────
CREATE POLICY "posts: 누구나 읽기" ON posts FOR SELECT USING (is_published = true);
CREATE POLICY "posts: 로그인 유저 생성" ON posts FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "posts: 본인만 수정" ON posts FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "posts: 본인만 삭제" ON posts FOR DELETE USING (auth.uid() = user_id);

-- ────────────────────────────────────────────
-- comments
-- ────────────────────────────────────────────
CREATE POLICY "comments: 누구나 읽기" ON comments FOR SELECT USING (true);
CREATE POLICY "comments: 로그인 유저 생성" ON comments FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "comments: 본인만 수정" ON comments FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "comments: 본인만 삭제" ON comments FOR DELETE USING (auth.uid() = user_id);

-- ────────────────────────────────────────────
-- notifications (본인 것만)
-- ────────────────────────────────────────────
CREATE POLICY "notifications: 본인만 읽기" ON notifications FOR SELECT USING (auth.uid() = user_id);
CREATE POLICY "notifications: 본인만 수정" ON notifications FOR UPDATE USING (auth.uid() = user_id);

-- ────────────────────────────────────────────
-- follows
-- ────────────────────────────────────────────
CREATE POLICY "follows: 누구나 읽기" ON follows FOR SELECT USING (true);
CREATE POLICY "follows: 본인만 생성" ON follows FOR INSERT WITH CHECK (auth.uid() = follower_id);
CREATE POLICY "follows: 본인만 삭제" ON follows FOR DELETE USING (auth.uid() = follower_id);

-- 나머지 테이블도 동일 패턴 적용 (본인 데이터만 CUD, 공개 데이터는 R)
```
